##Commands to run project:

#Command to create virtual environment

python -m venv .venv
 
#Command to activate virtual environment path

.\.venv\Scripts\activate

Or,

.\.venv\Scripts\Activate.ps1

#Install requirements

pip install -r requirements.txt

#Create a .env file, copy paste the following and replace with your api keys:

DATALAB_API_KEY=your_datalab_key_here
ANTHROPIC_API_KEY=your_anthropic_key_here

#To run main files for extraction and structure data

python main.py (pdf_file_name.pdf)

#To view the frontend

streamlit run viewer_streamlit.py
